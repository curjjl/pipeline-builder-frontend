function loadData() {
  if (!graph.value) return

  graph.value.freeze()

  // æ¸…ç©ºç”»å¸ƒ
  graph.value.clearCells()

  // æ·»åŠ èŠ‚ç‚¹

    const shape = getNodeShape(node.type)
    graph.value!.addNode({
      id: node.id,
      shape,
      x: node.x,
      y: node.y,
      label: node.label,
      data: node.data
    })
  })

  // æ·»åŠ è¾¹
  props.edges.forEach(edge => {
    graph.value!.addEdge({
      id: edge.id,
      source: { cell: edge.source, port: edge.sourcePort },
      target: { cell: edge.target, port: edge.targetPort },
      attrs: {
        line: {
          stroke: '#4A90E2',
          strokeWidth: 2,
          targetMarker: {
            name: 'block',
            width: 8,
            height: 8
          }
        }
      }
    })
  })

  graph.value.unfreeze()
  graph.value.centerContent()
}

function getNodeShape(type: string): string {
  if (type.includes('input') || type.includes('dataset')) return 'input-node'
  if (type.includes('output')) return 'output-node'
  return 'transform-node'
}

function nodeToData(node: any): Node {
  const position = node.position()
  return {
    id: node.id,
    type: node.data?.type || 'unknown',
    label: node.label as string,
    x: position.x,
    y: position.y,
    data: node.data
  }
}

function edgeToData(edge: any): Edge {
  return {
    id: edge.id,
    source: edge.getSourceCellId(),
    target: edge.getTargetCellId(),
    sourcePort: edge.getSourcePortId(),
    targetPort: edge.getTargetPortId()
  }
}
